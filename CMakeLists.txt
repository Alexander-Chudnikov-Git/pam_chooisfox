cmake_minimum_required(VERSION 3.21.1)

set(CMAKE_DISABLE_SOURCE_CHANGES ON)
set(CMAKE_DISABLE_IN_SOURCE_BUILD ON)

include(cmake/utils/build_protection.cmake)

set(PROJECT_NAME "pam_chooisfox")
project(${PROJECT_NAME})

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_RELEASE OFF)
set(CMAKE_UPX_COMPRESS OFF)

set(PAM_CHOOISFOX_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(PAM_CHOOISFOX_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(PAM_CHOOISFOX_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)

set(CMAKE_POSITION_INDEPENDENT_CODE ON)

include(cmake/platform/generic_builder.cmake)
